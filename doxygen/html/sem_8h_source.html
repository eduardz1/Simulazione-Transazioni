52/*
53 * Try to access the resource
54 * INPUT:
55 * - sem_id: the ID of the semaphore IPC object
56 * - sem_num: the position of the semaphore in the array
57 * RESULT
58 * - if the resource is available (semaphore value > 0), the semaphore
59 * is decremented by one
60 * - if the resource is not available (semaphore value == 0), the
61 * process is blocked until the resource becomes available again
62 * - the returned value is the same as the invoked semop
63 */
64int sem_reserve(int sem_id, int sem_num);

66/*
67 * Release the resource
68 * INPUT:
69 * - sem_id: the ID of the semaphore IPC object
70 * - sem_num: the position of the semaphore in the array
71 * RESULT:
72 * - the semaphore value is incremented by one. This may unblock some
73 * process
74 * - the returned value is the same as the invoked semop
75 */
76int sem_release(int sem_id, int sem_num);
